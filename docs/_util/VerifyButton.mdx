import { useState } from "react";

export const VerifyButton = ({checkExercise, feedback}) => {
  const cn = `verify-btn ${
    feedback !== null ? `verify-btn--${feedback ? "success" : "fail"}` : ``
  }`;
  return (
    <button
      type="button"
      className={cn}
      onClick={() => checkExercise()}
    >
      정답 확인
    </button>
  )
};
